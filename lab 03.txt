`id_walizki` int NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`pojemnosc` int UNSIGNED NOT NULL,

`id_wlasciciela` INT UNSIGNED,
FOREIGN KEY (id_wlasciciela) REFERENCES postac(id_postaci) ON DELETE CASCADE
);
